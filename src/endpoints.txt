-Account: 
	+Login:
		* send GET accounts/login/{
			"username": "huyfan",
			"password": "123"
		}
		
		* return {
			"message": "Login successful",
			"username": "huyfan",
			"fullname": "Phan Cong Huy",
			"email": "huyfanost@gmail.com",
			"phoneNum": "0943124423",
			"address": "Da Nang",
			"role": "user"
		}
		
	+Register/ Create account (use for admin): 
		* send POST accounts/create/ {
			"username": "newbie",
			"password": "156",
			"email": "bob@example.com",
			"fullname": "Bob",
			"phonenumber": "107897",
			"address": ""
			"role": "admin" // if is user, don't need to send this field
		}
		
		* return {
			"message": "Account created successfully"
		}
	
	+Forget password

	+View accounts (use for admin)
	+Create account
		*POST /admin/accounts/create/
			{
			  "username": "johndoe",
			  "password": "securepassword123",
			  "email": "johndoe@example.com",
			  "fullname": "John Doe",
			  "phonenumber": "1234567890",
			  "address": "123 Elm Street",
			  "role": "user"
			}
			
		return {
					"message": "Account created successfully"
				}
				
		return 404_error if some fields are existed
			{
				"errors": {
					"username": [
						"account with this username already exists."
					],
					"email": [
						"account with this email already exists."
					]
				}
			}
	+Search account
		search by username or fullname or email or phonenumber
		*GET /admin/accounts/search/?username=john
			params
			{
				"username" : "john",
				"email" : "example.com"
				"phonenumber"
				"fullname"
			}
		
		return 
		[
		  {
			"username": "johndoe",
			"email": "john.doe@example.com",
			"fullname": "Johnathan Doe",
			"phonenumber": "1234567890",
			"address": "123 Elm Street",
			"role": "user"
		  }
		]
	
	
	+Update account (use for admin)
		*PATCH /admin/accounts/update/
			params{
					"email": "john.doe@example.com",
					"fullname": "Johnathan Doe"
					"phonenumber",
					"username",...
			}
		return 
		{
			"message": "Account updated successfully"
		}
		
		return 400_error if some fields already existed
		{
			"errors": {
				"username": [
					"account with this username already exists."
				]
			}
		}
	
		
	+Delele account (use for admin)
		*DELETE /admin/accounts/delete/
			params
			{
				"username": "johndoe"
			}
			
		return 
		{
			"message": "Account deleted successfully"
		}
		
		return 404_error if account not found
		{
			"error": "Account not found"
		}
